def get_season(month):

def calc_confidence_interval(data, confidence=0.95):
    n = len(data)
    mean = np.mean(data)
    sem = stats.sem(data)  # Standard error of the mean
    margin = sem * stats.t.ppf((1 + confidence) / 2, n - 1)  # t-eloszlás használata
    return mean, mean - margin, mean + margin, n

# Kategorikus változók one-hot encoding-ja
bikes = pd.get_dummies(bikes, columns=['wday', 'month', 'season'], drop_first=True)

# Prediktorok kiválasztása (numerikus + one-hot encoded oszlopok)
predictors = ['cloud_cover', 'humidity', 'pressure', 'radiation', 'precipitation', 
              'snow_depth', 'sunshine', 'mean_temp', 'min_temp', 'max_temp', 
              'weekend'] + [col for col in bikes.columns if col.startswith('wday_') or col.startswith('month_') or col.startswith('season_')]
X = bikes[predictors]
y = bikes['bikes_hired']

# Ellenőrizzük az adattípusokat és NaN-okat
#print("\nData types of predictors:")
#print(X.dtypes)
#print("\nAny NaN in predictors:", X.isna().any().any())
#print("Any NaN in target:", y.isna().any())

# Hiányzó értékek kezelése (ha vannak)
X = X.fillna(X.mean())  # Numerikus oszlopok átlagával töltjük ki
y = y.fillna(y.mean())  # Célváltozó átlagával (bár ez ritka)

# Biztosítjuk, hogy minden numerikus legyen
X = X.astype(float)
y = y.astype(float)

# Konstans hozzáadása a regresszióhoz
X = sm.add_constant(X)

# 3. Regressziós modell illesztése
try:
    model = sm.OLS(y, X).fit()
except Exception as e:
    print("Error fitting model:", e)
    raise

# 4. Regressziós eredmények kiírása
